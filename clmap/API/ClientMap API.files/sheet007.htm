TPopupMenu начало…
результат имя тип параметр комментарий
public   TPopupMenu Создание экземпляра класса
      int width Ширина меню в пикселях
public void Set_zIndex Установка zIndex'a  
      int new_index  
public void setDiv Установка Div'a, куда выводится меню
      HTMLObject div  
public void AddItem Добавление пункта меню  
  string caption Текст пункта
  function onclick Данная функция вызывается при клике на пункте
  bool disabled пункт отключен или нет
      object sender Какой объект передается в функцию onclick
public void EnableItem Засеривание пункта меню
  int index Индекс элемента меню
      bool disabled пункт активен или нет
public int GetItemsCount() Получение количества пунктов в меню
с версии 1.2.0.2 и выше        
public void ClearItems() Удаление всех пунктов меню
с версии 1.2.0.2 и выше        
public void RemoveItem Удаление пункта меню  
      int index Индекс элемента меню
public void SetItemVisibility установка видимости пункта
  int index Индекс элемента меню
      bool visible видимость пункта
public int GetItemIndex Получение индекса пункта по тексту
      string caption  
TPopupMenu …продолжение
результат имя тип параметр комментарий
public void Hide() Закрытие меню  
public bool IsVisible() Проверка видимости меню
public void Popup Открытие меню  
  int x x, экранная координата
      int y y, экранная координата
public void Kill() Убиваем меню  
public void CheckItem() Установка флажка для пункта меню
с версии 1.2.0.36 и выше int index Индекс элемента меню
public void UnCheckItem() Снятие флажка для пункта меню
с версии 1.2.0.36 и выше int index Индекс элемента меню
public bool GetItemCheckState() Получение статуса флажка для пункта меню
с версии 1.2.0.36 и выше int index Индекс элемента меню
