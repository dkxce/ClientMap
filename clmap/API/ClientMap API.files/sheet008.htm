ClientCookie
результат имя тип параметр комментарий
public   ClientCookie() Создание экземпляра класса  
  string name имя кука
      string value значения кука (может отсутствовать)
public void set() Установка кука  
public void get() Получение значения кука  
public void del() Удаление кука    
public void DaysTimeout() Установка срока действие кука в днях
      int days дни
public void MonthTimeout() Установка срока действия кука в месяцах
      int mns месяцы

public string name Имя кука
